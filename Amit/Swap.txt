import java.util.Scanner;

class Swap{
	public static void main(String args[])
	{
		Scanner sc = new Scanner(System.in);
		System.out.println("Enter First Number:");
		int num1 = sc.nextInt();
		System.out.println("Enter Second Number:");
		int num2 = sc.nextInt();
		System.out.println("Given Numbers: \n num1 = "+num1+" \n num2 = "+num2);
		num1 = num1 + num2;
		num2 = num1 - num2;
		num1 = num1 - num2;
		System.out.println("Swapped Numbers: \n num1 = "+num1+" \n num2 = "+num2);
	}
}
