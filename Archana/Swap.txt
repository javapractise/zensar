package Practice;
import java.util.*;
public class Swap {
	public static void main(String args[]){
		int x,y,temp;
		Scanner sc=new Scanner(System.in);
		System.out.println("Enter the A and B values:");
		x=sc.nextInt();
		y=sc.nextInt();
		temp=x;
		x=y;
		y=temp;
		System.out.println("After swapping"+x+" "+y);
	}
}
