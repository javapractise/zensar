package Inheritance;

public class Triangle {
int a, b, c;
double getArea() {
	double s = (a+b+c)/2.0;
	return Math.pow((s*(s-a)*(s-b)*(s-c)), .5);
}
double getPerimeter() {
	return (a+b+c)/2.0;
}
	public static void main(String[] args) {
		Triangle t = new Triangle();
		t.a = 3;
		t.b = 4;
		t.c = 5;
		System.out.println(t.getArea());
		System.out.println(t.getPerimeter());		
	}

}
