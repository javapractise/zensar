	public static void main(String[] args) {

}


PALINDROME

package practice;

import java.util.Scanner;

public class palindrome {

	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int a=sc.nextInt();
		int sum=0;
		int rem=0;
		while(a!=0)
		{
			rem=a%10;
			sum=(sum*10)+rem;
			a=a/10;
		}
		System.out.println(sum)	;

	}

}


PRIME OR NOT

package practice;

import java.util.Scanner;

public class prime {
	
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int n=sc.nextInt();
		int count=0;
		for(int i=1;i<n;i++)
		{
			if(n%i==0)
			{
				count++;
			}
		}
		if(count==1)
		{
			System.out.print("prime");
		}
		else
		{
			System.out.print("not prime");
		}
			
			
	}

}


STRING REVERSE

package practice;

import java.util.Scanner;

public class stringreverse {

	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		String str=sc.next();
		int i=0;
		for( i=str.length()-1;i>=0;i--)
		{
			System.out.print(str.charAt(i));
		}
		

	}

}


SWAPING

package practice;

public class swaping {
	public static void main(String[] args){

		c=a;
		a=b;
