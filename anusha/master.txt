package serialization;



import java.io.FileOutputStream;
import java.io.ObjectOutputStream;



public class serialization {



public static void main(String[] args) {
try {
main obj=new main(12,"anusha",300);
FileOutputStream op=new FileOutputStream("D:anusha.txt");
ObjectOutputStream os=new ObjectOutputStream(op);
os.writeObject(obj);
os.flush();
os.close();
System.out.println("serialized");
}catch(Exception e)
{
System.out.println(e);
}



}



}

package serialization;



import java.io.FileInputStream;
import java.io.ObjectInputStream;



public class deserialization {



public static void main(String[] args) {
try {
main obj1=new main(13,"anusha",3000);
FileInputStream ip=new FileInputStream("D:anusha1.txt");
ObjectInputStream is=new ObjectInputStream(ip);
is.readObject();
System.out.println(obj1.name+""+obj1.roll_no+""+obj1.salary);
is.close();
}catch(Exception e) {
System.out.println(e);
}



}



}
