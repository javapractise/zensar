import java .util.*;
class ToyShop
{static int numofToys(int price[],
int N, int K)
{
int c = 0, sum = 0;

Arrays.sort(price);
for (int i = 0; i < N; i++)
{

if (sum +price[i] <= K)
{
sum = sum + price[i];

c++;
}
}
return c;
}public static void main (String[] args)
{
