package ExceptionHandling;
import java .io.*;
public class ReverseFibonacci {
	int n=sc.nextInt();
	Scanner sc = new Scanner(System.in);
	static void Display() {
	 int a[] = new int[n];
	 a[0]=0;
	 a[1]=1;
	 for(int i=2;i<n;i++)
	 {
		 a[i]= a[i-2]+a[i-1];
	 }
	 for (int i=n-1;i>=0;i--)
	 {
		 System.out.println(a[i]+  " ");
	 }
 }
	public static void main(String[] args) {
		int n=5;
		Display(n);
	}

}
