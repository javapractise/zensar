//is a serialization
package serialization;



import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.io.Serializable;



class args implements Serializable{
public int roll_no;
public String name;
public args(int roll_no, String name) {
super();
this.roll_no = roll_no;
this.name = name;
}



}




class student extends isaserialization {
public student(int roll_no, String name, String designation, int sal) {
super();
this.designation = designation;
this.sal = sal;
}
public String designation;
public int sal;

}




public class isaserialization {
public static void main(String args[]) {
try {
student s1=new student(15,"hemanth","jr.engineer",30000);
FileOutputStream os=new FileOutputStream("D:hemanth.txt");
ObjectOutputStream vi=new ObjectOutputStream(os);
vi.writeObject(s1);
vi.flush();
vi.close();
System.out.println("serialized");
}catch(Exception e) {
System.out.println(e);
}
try {
student s2=new student(16,"hemanth","jr.engineer",30000);
FileInputStream is=new FileInputStream("D:hemanth.txt");
ObjectInputStream as=new ObjectInputStream(is);
as.readObject();
as.close();
System.out.println("success");
}catch(Exception e) {
System.out.println(e);
}
}

}
