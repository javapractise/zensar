import java.util.*;  
class Swap_With {  
    public static void main(String[] args) {  
       int A, B, t;   
       Scanner sc = new Scanner(System.in);  
       System.out.println("Enter the value of A and B");  
       A = sc.nextInt();  
       B = sc.nextInt();  
       System.out.println("before swapping numbers: "+A +"  "+ B);  
        
       t = A;  
       A = B;  
       B = t;  
       System.out.println("After swapping: "+A +"   " + B);  
       System.out.println( );  
    }    
}
