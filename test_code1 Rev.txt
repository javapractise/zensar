package practice;
import java.util.*;
public class test1 {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int n=sc.nextInt();
		String s="";
		for(int i=0;i<n;i++){
			String w=sc.next();
			w+="";
			s+=w;
		}
		String[] a=s.split(" ");
		String s2="";
		for(int i=0;i<n;i++){
			if((a[i].length()%2!=0)&&(a[i].length()>s2.length()))
			s2=a[i];
		}
		if(s2.length()>0)
		System.out.println(s2);
		else
		System.out.println("Better luck next time");		
	}

}
